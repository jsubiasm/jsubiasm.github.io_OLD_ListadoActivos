	public static void main(String[] args) throws Exception {

		Map<Long, String> map = new HashMap<Long, String>();

		BufferedReader reader = new BufferedReader(new FileReader("C:/Users/Empleado/Desktop/tvc4/tvc4_ids1.txt"));

		String line = null;
		while ((line = reader.readLine()) != null)
		{
			try
			{
				map.put(Long.valueOf(line), "OK");
			}
			catch (Exception e)
			{
				System.out.println("Error al parsear [" + line + "]");
			}
		}
		reader.close();
		
		List<Long> lista = new ArrayList<Long>();
		lista.addAll(map.keySet());
		Collections.sort(lista);
		
		BufferedWriter writer = new BufferedWriter(new FileWriter("C:/Users/Empleado/Desktop/tvc4/tvc4_ids2.txt"));
		for (Iterator<Long> iterator = lista.iterator(); iterator.hasNext();)
		{
			Long id = (Long) iterator.next();
			writer.write("" + id + "\n");
		}
		writer.close();
		
    }
